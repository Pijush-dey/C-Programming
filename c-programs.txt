//Find out the Area of a Triangle-----------------------------------------------------<5>
#include <stdio.h>
#include <math.h>
int a_rea(int a, int b, int c);
int main()
{
    int a, b, c;
    printf("Enter the 1st length of the triangle:\n");
    scanf("%d", &a);
    printf("Enter the 2nd length of the triangle:\n");
    scanf("%d", &b);
    printf("Enter the 3rd length of the triangle:\n");
    scanf("%d", &c);
    printf("The area of the triangle is:%d\n", a_rea(a, b, c));
    return 0;
}
int a_rea(int a, int b, int c)
{
    if (a <= 0 || b <= 0 || c <= 0 || a + b <= c || b + c <= a || a + c <= b)
    {
        return 0;
    }
    else
    {
        int S = (a + b + c) / 2;
        return (sqrt(S * (S - a) * (S - b) * (S - c)));
    }
}

#include <stdio.h>
#include <math.h>
int main()
{
    long int n;
    int i, c = 0;
    printf("Enter the number:");
    scanf("%d", &n);
    printf("\n");
    if (n == 0 || n == 1)
    {
        printf("%d is neither prime nor composite number\n", n);
    }
    else
    {
        for (i = 2; i < sqrt(n); i++)
        {
            if (n % i == 0)
            {
                c = 1;
                break;
            }
        }
        if (c == 1)
        {
            printf("%d is a composite number\n", n);
        }
        else
        {
            printf("%d is a prime number\n", n);
        }
    }
    return 0;
}

{
    if (b == 0)

        return Arm_1;

    else
    {
        Arm_1 = Arm_1 + b % 10 * b % 10 * b % 10;

    }
}

{
    int b, Arm_1;

    Arm_1 = armstrong(b / 10, 0);

    {
        printf("%d is a Armstrong number\n", b);
    }
    else
    {
        printf("%d is not a Armstrong number\n", b);
    }
    return 0;
}

#include <stdio.h>
void star_pattern(int rows)
{
    for (int i = 1; i <= rows; i++)
    {
        for (int j = 1; j <= i; j++)
        {
            printf("*");
        }
        printf("\n");
    }
}
void reverse_star_pattern(int rows)
{
    for (int i = 0; i < rows; i++)
    {
        for (int j = 1; j <= (rows - i); j++)
        {
            printf("*");
        }
        printf("\n");
    }
}
int main()
{

#include <stdio.h>
#include <string.h>
void parser(char string[])
{
    int in = 0; //Variable weither we are in the tag or not
    int index = 0;
    for (int i = 0; i < strlen(string); i++)
    {
        if (string[i] == '<')
        {
            in = 1;
        }
        continue;
        else if (string[i] == '>')
        {
            in = 0;
        }
        if (in == 0)
        {
            string[index] = string[i + 1];
            index++;
        }
    }
    //Removing the spaces from the beginning
    while (string[0] == ' ')
    {
        for (i = 0; i < strlen(string); i++)
        {
            string[i] = string[i + 1];
        }
    }
    //Remove the spaces from the end of the string
    while (string[strlen(string) - 1] = ' ')
    {
        string[strlen(string) - 1] = '\0';
    }
}

{

    puts(string);

    float sum = 0.0;
    int *ptr;
    printf("Enter the no. of LOOPS: ");

    printf("\n\n");
    for (loopcount = 1; loopcount <= loops; loopcount++)
    {
        printf("HOw many numbers you need?");
        scanf("%d", &i);
        ptr = (float *)malloc(i * sizeof(float));
        printf("Enter the following numbers:\n");
        for (j = 1; j <= i; j++)
        {
            scanf("%4f\n", ptr[j]);
        }
        if (ptr[j] > 0)
        {
            sum += ptr[j];
            ++avg;
        }
        else
        {
            continue;
        }
        printf("Average of the non-zer numbers in the given list is: %3f", sum / avg);

}
