//Find out the Area of a Triangle-----------------------------------------------------<5>
#include <stdio.h>
#include <math.h>
int a_rea(int a, int b, int c);
int main()
{
    int a, b, c;
    printf("Enter the 1st length of the triangle:\n");
    scanf("%d", &a);
    printf("Enter the 2nd length of the triangle:\n");
    scanf("%d", &b);
    printf("Enter the 3rd length of the triangle:\n");
    scanf("%d", &c);
    printf("The area of the triangle is:%d\n", a_rea(a, b, c));
    return 0;
}
int a_rea(int a, int b, int c)
{
    if (a <= 0 || b <= 0 || c <= 0 || a + b <= c || b + c <= a || a + c <= b)
    {
        return 0;
    }
    else
    {
        int S = (a + b + c) / 2;
        return (sqrt(S * (S - a) * (S - b) * (S - c)));
    }
}

#include <stdio.h>
void star_pattern(int rows)
{
    for (int i = 1; i <= rows; i++)
    {
        for (int j = 1; j <= i; j++)
        {
            printf("*");
        }
        printf("\n");
    }
}
void reverse_star_pattern(int rows)
{
    for (int i = 0; i < rows; i++)
    {
        for (int j = 1; j <= (rows - i); j++)
        {
            printf("*");
        }
        printf("\n");
    }
}

#include <stdio.h>
#include <string.h>
void parser(char string[])
{
    int in = 0; //Variable weither we are in the tag or not
    int index = 0;
    for (int i = 0; i < strlen(string); i++)
    {
        if (string[i] == '<')
        {
            in = 1;
        }
        continue;
        else if (string[i] == '>')
        {
            in = 0;
        }
        if (in == 0)
        {
            string[index] = string[i + 1];
            index++;
        }
    }
    //Removing the spaces from the beginning
    while (string[0] == ' ')
    {
        for (i = 0; i < strlen(string); i++)
        {
            string[i] = string[i + 1];
        }
    }
    //Remove the spaces from the end of the string
    while (string[strlen(string) - 1] = ' ')
    {
        string[strlen(string) - 1] = '\0';
    }
}

    printf("\n");
    printf("1.Calculate the depreciation using Straight-line method !\n");
    printf("2.Calculate the depreciation using Double-declining balance method !\n");
    printf("3.Calculate the depreciation using Sum of the year's digits method !\n");
    printf("Enter your choice:");
    scanf("%d", &i);
    printf("\n");
    switch (i)
    {
    case 1:
        printf("Your choice is 1\n");

            amount -= dep;
            printf("Year no. %d", n);
            printf("\t Depreciation %d", dep);
            printf("\t Present AMOUNT %d", amount);
            printf("\n");
        }
        break;
    case 2:
        printf("Your choice is 2\n");
        for (n = 1; n <= year; n++)
        {
            dep = (2 / year) * amount;
            amount -= dep;
            printf("Year no. %d", n);
            printf("\t Depreciation %d", dep);
            printf("\t Present AMOUNT %d", amount);
            printf("\n");
        }
        break;
    case 3:
        printf("Your choice is 3\n");
        for (n = 1; n <= year; n++)
        {
            dep = (year - n + 1) / ((n * (n + 1)) / 2);

            printf("Year no. %d", n);
            printf("\t Depreciation %d", dep);
            printf("\t Present AMOUNT %d", amount);
            printf("\n");
        }
        break;
    default:

    }
    return 0;
}

#include <stdio.h>
#define MAX 50
    int main()
    {
        int a[MAX][MAX];
        int r;
        int zero = 0;
        int non_zero = 0;
        printf("Enter how many no. of rows you want:");
        scanf("%d", &r);
        printf("\n");
        for (int i = 0; i < r; i++)
        {
            for (int j = 0; j < r; j++)
            {
                printf("element of %2d-row & %2d-column:", i, j);
                scanf("%2d", &a[i][j]);
                printf("\n");
            }
        }
        for (int i = 0; i < r; i++)
        {
            for (int j = 0; j < r; j++)
            {
                if (a[i][j] == 0)
                    ++zero;
                else
                    ++non_zero;
            }
        }
        if (zero >= non_zero)
            printf("It is a sparse matrix!\n");
        else
            printf("It is not a sparse matrix!\n");
        return 0;
    }
